306 void init_32bits_timer()
307 {
308  T2CONbits.T32 = 1 ;/*TMRx and TMRy form a 32-bit timer*/
309  T2CONbits.TCKPS = 0 ;/*1:1 prescale value*/
310  T2CONbits.TCS = 0 ;/* Internal clock (FOSC/2)*/
311  T2CONbits.TGATE = 0 ;/*Gated time accumulation disabled*/
312  PR3 = 0xFFFF ;/*Définit les MSB de la période du timer*/
313  PR2 = 0xFFFF ;/*Définit les LSB de la période du timer*/
314  T2CONbits.TON = 1 ;/*Starts 32-bit TMRx:TMRy timer pair*/
315  /*The timer value at any point is stored in the register pair TMR3:TMR2. TMR3 always contains the most significant word of the count, while TMR2 contains the least significant word.*/
316  TMR3 = 0;
317  TMR2 = 0;
318 }
